\documentclass[tikz]{standalone}
\usepackage{pgfmath}
\usepackage{tikz-3dplot}
\usepgflibrary{shapes.callouts}
\usepgflibrary{shapes.misc}

\begin{document}

\tdplotsetmaincoords{70}{00}

\begin{tikzpicture}

\foreach \i/\l/\c in {0/1/blue,1/2/green!70!black,2/3/magenta,3/4/cyan,4/5/orange} {
	%\shade[ball color=orange,yshift=1cm]	(\i,0) circle (0.15);
	\node[yshift=1cm,rounded rectangle,text
		width=0.3cm,rotate=60,fill=\c,label=30:\l] (a\i) at (\i * 0.75,0) {};
}

\begin{scope}[every node/.style={circle,line width=2pt,draw=gray}]

\foreach \i/\c in {0/gray,1/gray,2/red,3/gray} {
	\node[yshift=0cm,xshift=0,fill=\c] (b\i) at (\i,0) {};
}

\foreach \i/\c in {0/gray,1/gray,2/red,3/gray,4/gray} {
	\node[yshift=-0.87cm,xshift=-0.5cm,fill=\c] (c\i) at (\i,0) {};
}

\foreach \i/\c in {0/gray,1/red,2/red,3/gray} {
	\node[yshift=-1.73cm,xshift=0,fill=\c] (d\i) at (\i,0) {};
}

\end{scope}

% annotations
\node[anchor=east] (drugs) at (-1,1) {drugs};
\node[anchor=east] (targets) at (-1,0) {targets};
\node[anchor=east] (genes) at (-1,-0.87) {genes};
\node[anchor=north,red] (AD genes) at (1.5,-2.25) {AD genes};

% arrows for annotations
\draw[->] (drugs) -- ($ (drugs) !0.75! (a0) $);
\draw[->] (targets) -- ($ (targets) !0.75! (b0) $);
\foreach \x  in {b,c,d} {
	\draw[->] (genes) -- ($ (genes) !0.75! (\x0) $);
}
\foreach \x  in {d1,c2,d2} {
	\draw[->,red] (AD genes) -- ($ (AD genes) !0.75! (\x) $);
}

% gene-gene edges
\path[line width=2pt,gray]
	% withn b's
	(b0) edge (b1)
	(b2) edge (b3)
	% withn c's
	(c0) edge (c1)
	(c2) edge (c3)
	(c3) edge (c4)
	% withn d's
	(d0) edge (d1)
	(d1) edge (d2)
	(d2) edge (d3)
	% between b's and c's
	(b0) edge (c0)
	(b0) edge (c1)
	(b1) edge (c1)
	(b2) edge (c2)
	(b3) edge (c2)
	(b3) edge (c3)
	(b3) edge (c4)
	% between c's and d's
	(c0) edge (d0)
	(c2) edge (d0)
	(c2) edge (d2)
	(c2) edge (d3)
	(c3) edge (d3)
	% between b's and d's
	(b3) edge (d3)
	;

% drug-gene edges
% a0: blue pill
\path[line width=2pt,blue]
	(a0) edge (b0)
	(a0) edge (b1)
	;

% a1: green pill
\path[line width=2pt,green!70!black]
	(a1) edge (b0)
	(a1) edge (b1)
	;

% a2: magenta pill
\path[line width=2pt,magenta]
	(a2) edge (b1)
	;

% a3: cyan pill
\path[line width=2pt,cyan]
	(a3) edge (b1)
	(a3) edge (b2)
	%(a3) edge (b3)
	;

% a4: orange pill
\path[line width=2pt,orange]
	(a4) edge (b3)
	;

\end{tikzpicture}

\end{document}
